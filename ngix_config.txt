server {
    listen 80;
    server_name schreinertool.de www.schreinertool.de hilbertcnc.de www.hilbertcnc.de;

    # HTTP → HTTPS
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl;
    server_name schreinertool.de www.schreinertool.de hilbertcnc.de www.hilbertcnc.de;

    ssl_certificate /etc/letsencrypt/live/c3cad.de/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/c3cad.de/privkey.pem;

    location / {
        proxy_pass http://localhost:3000;

        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
~
